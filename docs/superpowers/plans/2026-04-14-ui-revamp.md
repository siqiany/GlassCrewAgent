```python
def process_step(step):
    """
    Process each step from CrewAI and send events through SSE.
    """
    # Extract step information
    step_type = step.get("type") if isinstance(step, dict) else None
    
    # Get agent information
    agent = None
    if hasattr(step, 'agent') and step.agent:
        agent = getattr(step.agent, 'role', str(step.agent))
    elif isinstance(step, dict) and 'agent' in step:
        agent = step['agent']
    
    # Get action/tool information
    action = None
    tool_name = None
    tool_input = None
    tool_output = None
    
    if hasattr(step, 'action'):
        action = step.action
    if hasattr(step, 'tool_name'):
        tool_name = step.tool_name
    if hasattr(step, 'tool_input'):
        tool_input = step.tool_input
    if hasattr(step, 'tool_output'):
        tool_output = step.tool_output
    
    # Create event
    event = {
        'type': 'agent_step',
        'agent': agent,
        'action': action,
        'tool': tool_name,
        'input': str(tool_input) if tool_input else None,
        'output': str(tool_output) if tool_output else None,
        'timestamp': str(uuid.uuid4())[:8]
    }
    
    # Send to SSE queue
    task_info['stream_queue'].put(json.dumps(event))
```

- [ ] **Step 4: Add switchTab JavaScript function**

Add this in the `<script>` section after the other functions:

```javascript
// 切换标签
function switchTab(tabName) {
    // 移除所有active
    document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));
    
    // 添加active到选中标签
    if (tabName === 'log') {
        document.getElementById('tabLog').classList.add('active');
        document.getElementById('logTab').classList.add('active');
    } else {
        document.getElementById('tabResult').classList.add('active');
        document.getElementById('resultTab').classList.add('active');
    }
}
```
